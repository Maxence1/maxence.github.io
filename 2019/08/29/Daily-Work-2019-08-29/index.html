func getFPTOrderFare(foodQuoteInfo cartpb.Quote, deliveryQuoteInfo *cartpb.Quote, exponent int64) (commonpb.FPTOrderFare, error) {
if foodQuoteInfo == nil || deliveryQuoteInfo == nil {
return nil, pkgerrors.Wrap(errors.New(“getFPTOrderFare invalid param”), “”)
}

reducedPrice := cartDTO.PriceInMajor(foodQuoteInfo.ReducedPriceInMinorUnit+deliveryQuoteInfo.ReducedPriceInMinorUnit, int(exponent))
claimPrice := cartDTO.PriceInMajor(foodQuoteInfo.PriceInMinorUnit+deliveryQuoteInfo.PriceInMinorUnit, int(exponent))
tax := cartDTO.PriceInMajor(foodQuoteInfo.Tax+deliveryQuoteInfo.Tax, int(exponent))
subTotal := reducedPrice + tax
reducedPriceDiff := claimPrice - reducedPrice

// FPTOrderFare requires INT64, which may cause loss of precision
return &commonpb.FPTOrderFare{
    SubTotal:     int64(subTotal),
    Tax:          int64(tax),
    ReducedPrice: int64(reducedPriceDiff),
}, nil

}

func getFPTOrderFare(foodQuoteInfo cartpb.Quote, deliveryQuoteInfo *cartpb.Quote, exponent int64) (commonpb.FPTOrderFare, error) {
if foodQuoteInfo == nil || deliveryQuoteInfo == nil {
return nil, pkgerrors.Wrap(errors.New(“getFPTOrderFare invalid param”), “”)
}

reducedPrice := cartDTO.PriceInMajor(foodQuoteInfo.ReducedPriceInMinorUnit, int(exponent))
claimPrice := cartDTO.PriceInMajor(foodQuoteInfo.PriceInMinorUnit, int(exponent))
tax := cartDTO.PriceInMajor(foodQuoteInfo.Tax+deliveryQuoteInfo.Tax, int(exponent))
subTotal := reducedPrice + tax
reducedPriceDiff := claimPrice - reducedPrice

// FPTOrderFare requires INT64, which may cause loss of precision
return &commonpb.FPTOrderFare{
    SubTotal:     int64(subTotal),
    Tax:          int64(tax),
    ReducedPrice: int64(reducedPriceDiff),
}, nil

}
